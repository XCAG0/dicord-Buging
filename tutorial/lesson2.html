الدرس الثاني: أساسيات JavaScript

بعد أن تعلمنا كيفية إنشاء صفحة HTML وتنسيقها باستخدام CSS، سنتعلم الآن كيفية إضافة التفاعل إلى صفحتنا باستخدام JavaScript.

ما هو JavaScript؟

JavaScript هي لغة برمجة تسمح لك بإضافة ميزات تفاعلية إلى موقع الويب الخاص بك. يمكن استخدامها لإنشاء عناصر متحركة، والتحقق من صحة النماذج، وتحديث المحتوى بشكل ديناميكي، وغير ذلك الكثير.

كيفية إضافة JavaScript إلى صفحة HTML

هناك ثلاث طرق لإضافة JavaScript إلى صفحة HTML:

1. JavaScript داخلي (في عنصر script)

<script>
    // كود JavaScript هنا
    document.getElementById("demo").innerHTML = "مرحباً بالعالم!";
</script>

2. JavaScript خارجي (ملف منفصل)

أولاً، قم بإنشاء ملف JavaScript منفصل (مثل script.js) ثم قم بربطه بصفحة HTML:

<script src="script.js"></script>

3. JavaScript مضمن (في سمة للعنصر)

<button onclick="alert('مرحباً بالعالم!')">انقر هنا</button>

أساسيات JavaScript

المتغيرات

المتغيرات هي حاويات لتخزين البيانات:

// إعلان المتغيرات
let name = "أحمد";      // متغير محلي
const age = 25;         // ثابت لا يمكن تغييره
var country = "مصر";    // متغير قديم (يفضل استخدام let)

أنواع البيانات

// أنواع البيانات الأساسية
let text = "مرحباً";           // نص (String)
let number = 42;               // رقم (Number)
let isActive = true;           // منطقي (Boolean)
let person = {                 // كائن (Object)
    name: "محمد",
    age: 30
};
let colors = ["أحمر", "أخضر", "أزرق"];  // مصفوفة (Array)
let empty = null;              // فارغ (Null)
let notDefined;                // غير معرف (Undefined)

العمليات الحسابية

// العمليات الحسابية
let a = 10;
let b = 5;
let sum = a + b;        // 15 (جمع)
let diff = a - b;       // 5 (طرح)
let product = a * b;    // 50 (ضرب)
let quotient = a / b;   // 2 (قسمة)
let remainder = a % b;  // 0 (باقي القسمة)
let power = a ** b;     // 100000 (أس)

جمل الشرط

// جملة if
let hour = 15;

if (hour < 12) {
    console.log("صباح الخير");
} else if (hour < 18) {
    console.log("مساء الخير");
} else {
    console.log("مساء النور");
}

// العامل الثلاثي
let age = 20;
let status = (age >= 18) ? "بالغ" : "قاصر";

الحلقات التكرارية

// حلقة for
for (let i = 0; i < 5; i++) {
    console.log("الرقم: " + i);
}

// حلقة while
let i = 0;
while (i < 5) {
    console.log("الرقم: " + i);
    i++;
}

// حلقة for...of (للمصفوفات)
let colors = ["أحمر", "أخضر", "أزرق"];
for (let color of colors) {
    console.log(color);
}

الدوال

// تعريف دالة
function greet(name) {
    return "مرحباً " + name + "!";
}

// استدعاء الدالة
let message = greet("سارة");
console.log(message);  // يطبع: مرحباً سارة!

// دالة سهمية (Arrow Function)
const multiply = (a, b) => a * b;
console.log(multiply(4, 5));  // يطبع: 20

التفاعل مع HTML

الوصول إلى عناصر HTML

// الوصول إلى عنصر بواسطة المعرف
let element = document.getElementById("myId");

// الوصول إلى عناصر بواسطة اسم الوسم
let paragraphs = document.getElementsByTagName("p");

// الوصول إلى عناصر بواسطة اسم الفئة
let items = document.getElementsByClassName("item");

// الوصول باستخدام منتقي CSS
let element = document.querySelector("#myId");
let elements = document.querySelectorAll(".item");

تغيير محتوى HTML

// تغيير المحتوى النصي
document.getElementById("demo").textContent = "نص جديد";

// تغيير محتوى HTML
document.getElementById("demo").innerHTML = "<strong>نص جديد</strong>";

تغيير سمات HTML

// تغيير سمة
document.getElementById("myImage").src = "new-image.jpg";

// تغيير نمط CSS
document.getElementById("demo").style.color = "red";
document.getElementById("demo").style.fontSize = "20px";

الاستجابة للأحداث

// إضافة معالج حدث
document.getElementById("myButton").addEventListener("click", function() {
    alert("تم النقر على الزر!");
});

// أو بطريقة أخرى
function showMessage() {
    alert("تم النقر على الزر!");
}
document.getElementById("myButton").onclick = showMessage;

مثال عملي

لنقم بإنشاء مثال بسيط يوضح كيفية استخدام JavaScript للتفاعل مع صفحة HTML:

<!DOCTYPE html>
<html lang="ar" dir="rtl">
<head>
    <meta charset="UTF-8">
    <title>مثال على JavaScript</title>
    <style>
        body {
            font-family: Arial, sans-serif;
            text-align: center;
            margin: 20px;
        }
        
        .counter {
            font-size: 24px;
            margin: 20px 0;
        }
        
        button {
            padding: 10px 15px;
            margin: 5px;
            cursor: pointer;
        }
    </style>
</head>
<body>
    <h1>عداد بسيط</h1>
    <div class="counter" id="counter">0</div>
    <button id="incrementBtn">زيادة</button>
    <button id="decrementBtn">نقصان</button>
    <button id="resetBtn">إعادة تعيين</button>
    
    <script>
        // الحصول على العناصر
        const counterElement = document.getElementById("counter");
        const incrementBtn = document.getElementById("incrementBtn");
        const decrementBtn = document.getElementById("decrementBtn");
        const resetBtn = document.getElementById("resetBtn");
        
        // تهيئة قيمة العداد
        let count = 0;
        
        // تحديث العرض
        function updateDisplay() {
            counterElement.textContent = count;
            
            // تغيير اللون بناءً على القيمة
            if (count > 0) {
                counterElement.style.color = "green";
            } else if (count < 0) {
                counterElement.style.color = "red";
            } else {
                counterElement.style.color = "black";
            }
        }
        
        // إضافة معالجات الأحداث
        incrementBtn.addEventListener("click", function() {
            count++;
            updateDisplay();
        });
        
        decrementBtn.addEventListener("click", function() {
            count--;
            updateDisplay();
        });
        
        resetBtn.addEventListener("click", function() {
            count = 0;
            updateDisplay();
        });
    </script>
</body>
</html>

كيف يعمل هذا المثال:

  1. قمنا بإنشاء عداد بسيط مع ثلاثة أزرار: زيادة، نقصان، وإعادة تعيين.
  2. استخدمنا JavaScript للوصول إلى عناصر HTML وتخزينها في متغيرات.
  3. أضفنا معالجات أحداث للأزرار لتغيير قيمة العداد عند النقر عليها.
  4. قمنا بإنشاء دالة updateDisplay() لتحديث العرض وتغيير لون النص بناءً على قيمة العداد.

تمرين عملي:

حاول توسيع المثال السابق بإضافة المزيد من الميزات مثل:

  • إضافة زر لمضاعفة القيمة الحالية
  • إضافة حقل إدخال للسماح للمستخدم بتعيين قيمة محددة
  • إضافة تأثيرات بصرية عند تغيير القيمة (مثل الرسوم المتحركة)
  • تخزين القيمة في التخزين المحلي (localStorage) بحيث تظل موجودة حتى بعد تحديث الصفحة
